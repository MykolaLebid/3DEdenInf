542 typedef struct tree {
543 
544  /* An array of pointers to nodes. Each tip node and ring of nodes has a
545  * unique index starting from one. The nodep array contains pointers to each
546  * one, starting from 0. In the case of internal nodes, the entries in nodep
547  * point to the rootward node in the group. Since the trees are otherwise
548  * entirely symmetrical, except at the root, this is the only way to resolve
549  * parent, child, and sibling relationships.
550  *
551  * Indices in range [0, spp) point to tips, while indices [spp, nonodes)
552  * point to fork nodes
553  */
554  pointarray nodep;
555 
556  /* A pointer to the first node. Typically, root is used when the tree is rooted,
557  * and points to an internal node with no back link. */
558  node *root;
559 
560  /* start is used when trees are unrooted. It points to an internal node whose
561  * back link typically points to the outgroup leaf. */
562  node *start;
563 
564  /* In maximum likelihood programs, the most recent evaluation is stored here */
565  double likelihood;
566 
567  /* Branch transition matrices for restml */
568  transptr trans; /* all transition matrices */
569  long *freetrans; /* an array of indexes of free matrices */
570  long transindex; /* index of last valid entry in freetrans[] */
571 } tree;
